Takes a remote (map) of settings and script a shell script file to execute it on the remote node and returns output. See docs at sshScript for more information.

    node {
      def remote = [:]
      remote.name = 'test'
      remote.host = 'test.domain.com'
      remote.user = 'root'
      remote.password = 'password'
      remote.allowAnyHosts = true
      stage('Remote SSH') {
        writeFile file: 'abc.sh', text: 'ls -lrt'
        sshScript remote: remote, script: "abc.sh"
      }
    }
