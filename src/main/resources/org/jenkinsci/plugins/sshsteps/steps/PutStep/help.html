Takes a remote (map) of settings, local file/directory from workspace and path to put this into remote node. See docs at sshPut for more information.

    node {
      def remote = [:]
      remote.name = 'test'
      remote.host = 'test.domain.com'
      remote.user = 'root'
      remote.password = 'password'
      remote.allowAnyHosts = true
      stage('Remote SSH') {
        writeFile file: 'abc.sh', text: 'ls -lrt'
        sshPut remote: remote, from: 'abc.sh', into: '.'
      }
    }
