Takes a remote (map) of settings and path (file/directory) to remove from remote node. See docs at sshRemove for more information.

    node {
      def remote = [:]
      remote.name = 'test'
      remote.host = 'test.domain.com'
      remote.user = 'root'
      remote.password = 'password'
      remote.allowAnyHosts = true
      stage('Remote SSH') {
        sshRemove remote: remote, path: "abc.sh"
      }
    }
